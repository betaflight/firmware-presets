# Title: Cinematic RC smoothing
# FirmwareVersion: 4.3
# Category: RC_SMOOTHING
# Official: true
# Keywords: cinematic, rc, smooth

# Description: RC smoothing settings for cinematic HD footage
# Description: May feel a bit 'soft' and un-responsive for small stick inputs
# Description: Should give a consistent feel across a wide range of RC links
# Description: Best used with Actual Rates at very low centre sensitivity with zero expo
# Description: There is no need to turn off feedforward
# Description: Reduce the rc_smoothing_setpoint_cutoff and rc_smoothing_feedforward_cutoff values if more smoothing is required

set rc_smoothing = ON
set feedforward_jitter_reduction = 20
set feedforward_transition = 0
set rc_smoothing_setpoint_cutoff = 13
set rc_smoothing_feedforward_cutoff = 13
set rc_smoothing_throttle_cutoff = 30
