# Title: Freestyle RC smoothing
# FirmwareVersion: 4.3
# Category: RC_SMOOTHING
# Official: true
# Keywords: cinematic, rc, smooth
# Author: Chris Thompson ctzsnooze
# Description: RC smoothing settings for Freestyle HD footage
# Description: Fixed cutoff frequency for consistent results over a wide range of RC link frequencies - the auto cutoff values are not used

# Description: There is no need to turn off feedforward or to enable transition
# Description: The jitter reduction provides dynamic attenuation of feedforward when sticks move slowly

# Description: Actual Rates, centre sensitivity around 100 with zero expo are recommended

set rc_smoothing = ON
set feedforward_jitter_reduction = 15
set feedforward_transition = 0
set rc_smoothing_setpoint_cutoff = 40
set rc_smoothing_feedforward_cutoff = 40
set rc_smoothing_auto_factor_throttle = 30
set rc_smoothing_throttle_cutoff = 0
