# Title: Race RC smoothing
# FirmwareVersion: 4.3
# Category: RC_SMOOTHING
# Official: true
# Keywords: aggressive, rc, smoothing, race
# Author: Chris Thompson ctzsnooze
# Description: RC smoothing settings for racing - assumes a good RC link
# Description: Slightly increases stick responsiveness over defaults
# Description: Note: defaults are close to ideal for racing
# Description: Warning: sends some noise to the motors at link frequency during fast stick movements - may make motors warmer than usual

set rc_smoothing = ON
set feedforward_jitter_reduction = 5
set feedforward_transition = 0
set rc_smoothing_auto_factor = 22
set rc_smoothing_auto_factor_throttle = 20
set rc_smoothing_setpoint_cutoff = 0
set rc_smoothing_feedforward_cutoff = 0
set rc_smoothing_throttle_cutoff = 0
