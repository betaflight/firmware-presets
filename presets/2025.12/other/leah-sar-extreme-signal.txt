#$ DESCRIPTION: Configures low-resource logging, aggressive OSD warnings
#$ DESCRIPTION: for signal degradation, and GPS rescue for maritime/high-interference environments.
#$ DISCUSSION: https://github.com/betaflight/firmware-presets/pull/570
#$ WARNING: Requires GPS module. Failsafe will trigger GPS rescue mode.
#$ DISCLAIMER: Intended for experienced pilots in SAR/field operations.

# --- LOW RESOURCE LOGGING ---
set blackbox_mode = NORMAL
set blackbox_rate_num = 1
set blackbox_rate_denom = 32

# --- LINK QUALITY (LQ) & RSSI OSD WARNINGS ---
set osd_rssi_alarm = 35
set osd_link_quality_alarm = 50
set osd_rssi_dbm_alarm = -100

# --- VTX SETTINGS ---
# VTX set to Raceband 4 (5769) for baseline team standardization 
# and separation from coastal Wi-Fi. Adjust for local regulations.
set vtx_low_power_disarm = ON
set vtx_freq = 5769


# --- FAILSAFE BEHAVIOR (GPS RESCUE FOR MARITIME/SAR) ---
set failsafe_procedure = RESCUE
set failsafe_delay = 15

# --- GPS RESCUE PARAMETERS ---
set gps_rescue_min_dth = 30
set gps_rescue_return_alt = 100

save
